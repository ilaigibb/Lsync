#!/bin/bash

set -e

VERSION="1.1.0"

SCRIPT_PATH="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
if [[ "$SCRIPT_PATH" != "/usr/local/bin" ]]; then
    CLONE_MARKER="[CLONE]"
else
    CLONE_MARKER=""
fi

RED='\033[0;31m'
GREEN='\033[0;32m'
CYAN='\033[0;36m'
YELLOW='\033[1;33m'
NC='\033[0m'

should_use_sudo() {
    [[ "$USE_SUDO" == true ]] && return 0
    [[ "$AUTO_SUDO" == true ]] && return 0
    return 1
}

get_rsync_path() {
    if should_use_sudo; then
        echo "sudo rsync"
    else
        echo "rsync"
    fi
}

get_remote_user() {
    if [[ -n "$REMOTE_USER" ]]; then
        echo "$REMOTE_USER"
    elif should_use_sudo; then
        echo "root"
    else
        echo "$USER"
    fi
}

prompt_sudo_if_needed() {
    if should_use_sudo; then
        return 0
    fi
    
    if [[ "$AUTO_SUDO" == true ]]; then
        USE_SUDO=true
        return 0
    fi
    
    return 0
}

build_rsync_filters() {
    local filters=""
    
    for pattern in "${EXCLUDE_PATTERNS[@]}"; do
        filters="$filters --exclude=$pattern"
    done
    
    if [[ -n "$MIN_SIZE" ]]; then
        filters="$filters --min-size=$MIN_SIZE"
    fi
    
    if [[ -n "$NEWER_THAN" ]]; then
        filters="$filters --newer-than=$NEWER_THAN"
    fi
    
    if [[ "$USE_CHECKSUM" == true ]]; then
        filters="$filters --checksum"
    fi
    
    echo "$filters"
}

is_color_enabled() {
    if [[ "$COLOR_OUTPUT" == false ]]; then
        return 1
    fi
    if [[ -n "$NO_COLOR" ]]; then
        return 1
    fi
    return 0
}

validate_config() {
    local has_warnings=0
    
    if [[ -z "$HOST" ]]; then
        echo -e "${RED}Error: HOST is not configured${NC}" >&2
        has_warnings=1
    fi
    
    if [[ -z "$USER" ]]; then
        echo -e "${RED}Error: USER is not configured${NC}" >&2
        has_warnings=1
    fi
    
    if [[ -n "$SSH_KEY" && ! -f "$SSH_KEY" ]]; then
        echo -e "${YELLOW}Warning: SSH key not found: $SSH_KEY${NC}" >&2
        has_warnings=1
    fi
    
    if [[ "$AUTO_SUDO" == true ]]; then
        echo -e "${YELLOW}Warning: AUTO_SUDO is enabled. Will auto-use sudo without prompting.${NC}" >&2
    fi
    
    if [[ $has_warnings -eq 1 ]]; then
        return 1
    fi
    return 0
}

DEFAULT_HOST="129.159.130.127"
DEFAULT_USER="ubuntu"
DEFAULT_SSH_KEY="$HOME/.ssh/id_rsa.key"
DEFAULT_REMOTE_PATH=".lodestone"
DEFAULT_LOCAL_PATH="$HOME/localServer"
DEFAULT_BACKUP_DIR="$HOME/.lsync_backups"
DEFAULT_PUSH_CMD="dpush"
DEFAULT_PULL_CMD="dpull"

HOST="$DEFAULT_HOST"
USER="$DEFAULT_USER"
SSH_KEY="$DEFAULT_SSH_KEY"
REMOTE_PATH="$DEFAULT_REMOTE_PATH"
LOCAL_PATH="$DEFAULT_LOCAL_PATH"
BACKUP_DIR="$DEFAULT_BACKUP_DIR"
PUSH_CMD="$DEFAULT_PUSH_CMD"
PULL_CMD="$DEFAULT_PULL_CMD"

QUIET=false
AUTO_YES=false
AUTO_SUDO=false
USE_SUDO=false
DEBUG=false
DRY_RUN=false
DELETE_FLAG=false
DEFAULT_REMOTE_USER=""
REMOTE_USER=""
EXCLUDE_PATTERNS=()
MIN_SIZE=""
NEWER_THAN=""
BACKUP_DAYS=30
COLOR_OUTPUT=true
VERBOSE=0
USE_CHECKSUM=false

CONFIG_DIR="${XDG_CONFIG_HOME:-$HOME/.config}/lsync"
CONFIG_FILE="$CONFIG_DIR/config"
PROFILES_DIR="$CONFIG_DIR/profiles"
DEFAULT_PROFILE=""

mkdir -p "$CONFIG_DIR" "$PROFILES_DIR"

load_config() {
    if [[ -f "$CONFIG_FILE" ]]; then
        source "$CONFIG_FILE"
    fi
    REMOTE_USER="${DEFAULT_REMOTE_USER:-$USER}"
}

init_config() {
    echo -e "${CYAN}${CLONE_MARKER}=== First Time Setup ===${NC}"
    echo ""
    
    echo -n "Enter SSH Host [$DEFAULT_HOST]: "
    read -r host_input
    HOST="${host_input:-$DEFAULT_HOST}"
    
    echo -n "Enter SSH User [$DEFAULT_USER]: "
    read -r user_input
    USER="${user_input:-$DEFAULT_USER}"
    
    echo -n "SSH Key Path [$DEFAULT_SSH_KEY]: "
    read -r key_input
    SSH_KEY="${key_input:-$DEFAULT_SSH_KEY}"
    
    echo -n "Local Path [$DEFAULT_LOCAL_PATH]: "
    read -r local_input
    LOCAL_PATH="${local_input:-$DEFAULT_LOCAL_PATH}"
    
    echo -n "Remote Path [$DEFAULT_REMOTE_PATH]: "
    read -r remote_input
    REMOTE_PATH="${remote_input:-$DEFAULT_REMOTE_PATH}"
    
    echo -n "Backup Directory [$DEFAULT_BACKUP_DIR]: "
    read -r backup_input
    BACKUP_DIR="${backup_input:-$DEFAULT_BACKUP_DIR}"
    
    echo -n "Push Command Name [$DEFAULT_PUSH_CMD]: "
    read -r push_input
    PUSH_CMD="${push_input:-$DEFAULT_PUSH_CMD}"
    
    echo -n "Pull Command Name [$DEFAULT_PULL_CMD]: "
    read -r pull_input
    PULL_CMD="${pull_input:-$DEFAULT_PULL_CMD}"
    
    echo -n "Auto-sudo for remote commands? (y/n) [n]: "
    read -r sudo_input
    [[ "$sudo_input" == "y" ]] && AUTO_SUDO=true || AUTO_SUDO=false
    
    echo -n "Default remote user (leave empty for current) []: "
    read -r user_input
    DEFAULT_REMOTE_USER="${user_input:-}"
    
    echo -n "Backup rotation (days, 0 to disable) [30]: "
    read -r backup_days_input
    BACKUP_DAYS="${backup_days_input:-30}"
    
    mkdir -p "$(dirname "$CONFIG_FILE")"
    
    cat > "$CONFIG_FILE" << EOF
# lsync configuration
# Generated by lsync init

HOST="$HOST"
USER="$USER"
SSH_KEY="$SSH_KEY"
REMOTE_PATH="$REMOTE_PATH"
LOCAL_PATH="$LOCAL_PATH"
BACKUP_DIR="$BACKUP_DIR"
PUSH_CMD="$PUSH_CMD"
PULL_CMD="$PULL_CMD"
AUTO_SUDO=$AUTO_SUDO
DEFAULT_REMOTE_USER="$DEFAULT_REMOTE_USER"
BACKUP_DAYS=$BACKUP_DAYS
EOF

    echo ""
    echo -e "${GREEN}${CLONE_MARKER}Config saved to $CONFIG_FILE${NC}"
    echo ""
    echo "You can edit this file anytime or run 'lsync init --reset' to reconfigure."
}

    echo ""
    echo "Options:"
    echo "  -y, --yes      Auto-confirm all prompts (including sudo)"

    echo "  --newer-than   Only sync files newer than time (e.g., 24h, 7d)"
    echo "  --no-colors    Disable colored output"
    echo "  -v, --verbose Increase verbosity (use multiple times for more detail)"

    echo "  remove <profile> [path-name]       Remove profile or path"
    echo "  profiles           List all profiles"
    echo "  default <profile>  Set default profile"
    echo "  rename <old> <new>  Rename profile"

    echo "  --version          Show version"
    echo ""
    echo "Config: $CONFIG_DIR/"
}

parse_flags() {
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -q|--quiet) QUIET=true; shift ;;
            -y|--yes) AUTO_YES=true; shift ;;
            --sudo) USE_SUDO=true; shift ;;
            -u|--user) REMOTE_USER="$2"; shift 2 ;;
            --debug) DEBUG=true; shift ;;
            --dry-run) DRY_RUN=true; shift ;;
            --delete) DELETE_FLAG=true; shift ;;
            --exclude) EXCLUDE_PATTERNS+=("$2"); shift 2 ;;
            --min-size) MIN_SIZE="$2"; shift 2 ;;
            --newer-than) NEWER_THAN="$2"; shift 2 ;;
            --no-colors) COLOR_OUTPUT=false; shift ;;
            -v|--verbose) VERBOSE=$((VERBOSE + 1)); shift ;;
            --checksum) USE_CHECKSUM=true; shift ;;
            -h|--help)
                show_help
                exit 0
                ;;
            --version)
                echo "lsync v$VERSION${CLONE_MARKER:+ $CLONE_MARKER}"
                exit 0
                ;;
            -*)
                shift
                ;;
            *)
                break
                ;;
        esac
    done
    
    CMD_ACTION="${1:-}"
    CMD_ARG="${2:-}"
}

RSYNC_PREVIEW_OPTS="-avzn --out-format=%i%n"
RSYNC_OPTS="-avz --progress"

if [[ "$DELETE_FLAG" == true ]]; then
    RSYNC_OPTS="$RSYNC_OPTS --delete"
fi

if [[ "$DRY_RUN" == true ]]; then
    RSYNC_PREVIEW_OPTS="$RSYNC_PREVIEW_OPTS -n"
fi

get_changes() {
    local src="$1"
    local dst="$2"
    local quiet="${3:-false}"
    
    local added=0
    local deleted=0
    local files=()
    local rsync_path
    rsync_path=$(get_rsync_path)
    local remote_user
    remote_user=$(get_remote_user)
    local dst_with_user
    local filters
    filters=$(build_rsync_filters)
    
    if [[ "$dst" == *@*:* ]]; then
        dst_with_user="${dst%%:*}:${dst#*:}"
        dst_with_user="${remote_user}@${dst_with_user#*@}"
    else
        dst_with_user="$dst"
    fi
    
    if [[ "$DEBUG" == true ]]; then
        echo "Running: rsync $RSYNC_PREVIEW_OPTS $filters --rsync-path=\"$rsync_path\" -e \"ssh -i $SSH_KEY -o StrictHostKeyChecking=no\" \"$src/\" \"$dst_with_user/\""
    fi
    
    rsync_output=$(rsync $RSYNC_PREVIEW_OPTS $filters --rsync-path="$rsync_path" -e "ssh -i $SSH_KEY -o StrictHostKeyChecking=no" "$src/" "$dst_with_user/" 2>&1)
    
    if [[ "$DEBUG" == true ]]; then
        echo "--- RSYNC OUTPUT START ---"
        echo "$rsync_output"
        echo "--- RSYNC OUTPUT END ---"
    fi
    
    rsync_output=$(echo "$rsync_output" | grep -v '^sending' | grep -v '^receiving' | grep -v '^total:' | grep -v '^Number of' | grep -v '^--' | grep -v '^sent ' | grep -v '^total ')
    
    while IFS= read -r line; do
        [[ -z "$line" ]] && continue
        indicator="${line:0:1}"
        [[ "$indicator" != ">" && "$indicator" != "<" && "$indicator" != "c" && "$indicator" != "d" ]] && continue
        filename="${line:11}"
        
        [[ -z "$filename" ]] && continue
        
        local is_dir=false
        if [[ "$filename" == */ ]]; then
            is_dir=true
            filename="${filename%/}"
        fi
        
        if [[ "$indicator" == "<" ]]; then
            ((deleted++))
            if [[ "$is_dir" == true ]]; then
                files+=("-${filename}/")
            else
                files+=("-${filename}")
            fi
        elif [[ "$indicator" == ">" || "$indicator" == "c" ]]; then
            ((added++))
            if [[ "$is_dir" == true ]]; then
                files+=("+${filename}/")
            else
                files+=("+${filename}")
            fi
        fi
    done <<< "$rsync_output"
    
    if [[ $added -eq 0 && $deleted -eq 0 ]]; then
        return 1
    fi
    
    if [[ "$quiet" == "true" || "$QUIET" == "true" ]]; then
        return 0
    fi
    
    echo -e "${GREEN}+$added${NC} ${RED}-$deleted${NC}:"
    for f in "${files[@]}"; do
        if [[ "$f" == -* ]]; then
            printf "${RED}$f${NC}\n" "${f:1}"
        else
            printf "${GREEN}$f${NC}\n"
        fi
    done
    
    return 0
}

confirm() {
    if [[ "$AUTO_YES" == true ]]; then
        return 0
    fi
    
    local prompt="$1"
    while true; do
        read -p "$prompt [y/n] " 
        case "$REPLY" in
            y|Y) return 0 ;;
            n|N) return 1 ;;
        esac
    done
}

backup_create() {
    local src="$1"
    local name="$2"
    local backup_path="$BACKUP_DIR/${name}_$(date +%Y%m%d_%H%M%S)"
    
    local rsync_path
    rsync_path=$(get_rsync_path)
    local remote_user
    remote_user=$(get_remote_user)
    local src_with_user
    
    if [[ "$src" == *@*:* ]]; then
        src_with_user="${remote_user}@${src#*@}"
    else
        src_with_user="$src"
    fi
    
    mkdir -p "$BACKUP_DIR"
    if [[ "$QUIET" != "true" ]]; then
        echo "Creating backup: $backup_path"
    fi
    
    local rsync_cmd
    if [[ "$src" == *@*:* ]]; then
        rsync_cmd="rsync -avq --rsync-path=\"$rsync_path\" -e \"ssh -i $SSH_KEY -o StrictHostKeyChecking=no\""
    else
        rsync_cmd="rsync -avq"
    fi
    
    if ! eval "$rsync_cmd" "$src_with_user" "$backup_path/" 2>/dev/null; then
        echo -e "${RED}${CLONE_MARKER}Error: Backup failed${NC}" >&2
        return 1
    fi
    if [[ "$QUIET" != "true" ]]; then
        echo "Backup created: $backup_path"
    fi
    echo "$backup_path" > "$BACKUP_DIR/latest_$name"
}

backup_restore() {
    local name="$1"
    local backup_path
    
    if [[ -z "$name" ]]; then
        echo "Usage: $0 restore <backup_name>"
        return 1
    fi
    
    if [[ -f "$BACKUP_DIR/$name" ]]; then
        backup_path=$(cat "$BACKUP_DIR/$name")
    else
        backup_path="$BACKUP_DIR/$name"
    fi
    
    if [[ ! -d "$backup_path" ]]; then
        echo "Backup not found: $backup_path"
        return 1
    fi
    
    echo "Backup: $backup_path"
    ls -la "$backup_path"
    
    if confirm "Restore this backup?"; then
        echo "Restore functionality coming soon"
    else
        echo "Restore cancelled."
    fi
}

backup_list() {
    echo "Available backups:"
    ls -lt "$BACKUP_DIR" 2>/dev/null | head -20
}

backup_rotate() {
    if [[ -z "$BACKUP_DAYS" || "$BACKUP_DAYS" -eq 0 ]]; then
        return 0
    fi
    
    local count=0
    while IFS= read -r backup; do
        [[ -z "$backup" ]] && continue
        rm -rf "$backup"
        ((count++))
    done < <(find "$BACKUP_DIR" -maxdepth 1 -type d -name "*_"* -mtime +"$BACKUP_DAYS" 2>/dev/null)
    
    if [[ $count -gt 0 ]]; then
        echo -e "${CYAN}Rotated $count old backup(s)${NC}"
    fi
}

profile_create() {
    local name="$1"
    
    if [[ -z "$name" ]]; then
        echo "Usage: lsync create <name>"
        return 1
    fi
    
    if [[ -f "$PROFILES_DIR/$name.conf" ]]; then
        echo -e "${YELLOW}${CLONE_MARKER}Profile '$name' already exists${NC}"
        return 1
    fi
    
    cat > "$PROFILES_DIR/$name.conf" << EOF
# lsync profile: $name
# Add paths using: lsync add $name <path-name> [folder]

EOF
    
    echo -e "${GREEN}${CLONE_MARKER}Created profile '$name'${NC}"
    echo "Add paths with: lsync add $name <path-name> [folder]"
}

profile_add() {
    local profile="$1"
    local path_name="$2"
    local folder="${3:-}"
    
    if [[ -z "$profile" || -z "$path_name" ]]; then
        echo "Usage: lsync add <profile> <path-name> [folder]"
        return 1
    fi
    
    local profile_file="$PROFILES_DIR/$profile.conf"
    
    if [[ ! -f "$profile_file" ]]; then
        echo -e "${RED}${CLONE_MARKER}Profile '$profile' does not exist${NC}"
        echo "Create it with: lsync create $profile"
        return 1
    fi
    
    local cwd
    local local_path
    local remote_path
    
    if [[ -z "$folder" ]]; then
        cwd="$(pwd)"
        local_path="$cwd"
        remote_path="$cwd"
    else
        cwd="$(pwd)"
        local_path="$cwd/$folder"
        remote_path="$cwd/$folder"
    fi
    
    echo "" >> "$profile_file"
    echo "[$path_name]" >> "$profile_file"
    echo "local=\"$local_path\"" >> "$profile_file"
    echo "remote=\"$remote_path\"" >> "$profile_file"
    
    echo -e "${GREEN}${CLONE_MARKER}Added '$path_name' to profile '$profile'${NC}"
    echo "  local: $local_path"
    echo "  remote: $remote_path"
}

profile_remove() {
    local profile="$1"
    local path_name="$2"
    
    local profile_file="$PROFILES_DIR/$profile.conf"
    
    if [[ ! -f "$profile_file" ]]; then
        echo -e "${RED}${CLONE_MARKER}Profile '$profile' does not exist${NC}"
        return 1
    fi
    
    if [[ -z "$path_name" ]]; then
        if confirm "Delete entire profile '$profile'?"; then
            rm "$profile_file"
            if [[ "$DEFAULT_PROFILE" == "$profile" ]]; then
                sed -i 's/DEFAULT_PROFILE=.*/DEFAULT_PROFILE=""/' "$CONFIG_FILE" 2>/dev/null || true
            fi
            echo -e "${GREEN}${CLONE_MARKER}Deleted profile '$profile'${NC}"
        else
            echo "Cancelled."
        fi
        return
    fi
    
    local tmp_file
    tmp_file=$(mktemp)
    
    local in_section=false
    while IFS= read -r line; do
        if [[ "$line" == "[$path_name]" ]]; then
            in_section=true
            continue
        fi
        if [[ "$in_section" == true && "$line" == \[\]*"]" ]]; then
            in_section=false
        fi
        if [[ "$in_section" == false ]]; then
            echo "$line"
        fi
    done < "$profile_file" > "$tmp_file"
    
    mv "$tmp_file" "$profile_file"
    echo -e "${GREEN}${CLONE_MARKER}Removed '$path_name' from profile '$profile'${NC}"
}

profile_list() {
    echo "Profiles:"
    if [[ -z "$(ls -A "$PROFILES_DIR" 2>/dev/null)" ]]; then
        echo "  No profiles yet. Create one with: lsync create <name>"
        return
    fi
    
    for profile_file in "$PROFILES_DIR"/*.conf; do
        [[ -f "$profile_file" ]] || continue
        local name
        name=$(basename "$profile_file" .conf)
        
        if [[ "$name" == "$DEFAULT_PROFILE" ]]; then
            echo -e "  ${GREEN}$name${NC} (default)"
        else
            echo "  $name"
        fi
        
        source "$profile_file" 2>/dev/null || continue
        
        local paths=()
        while IFS= read -r line; do
            if [[ "$line" == \[*] ]]; then
                local p="${line#[}"
                p="${p%]}"
                paths+=("$p")
            fi
        done < "$profile_file"
        
        for p in "${paths[@]}"; do
            echo "    - $p"
        done
    done
}

profile_set_default() {
    local profile="$1"
    
    if [[ -z "$profile" ]]; then
        if [[ -n "$DEFAULT_PROFILE" ]]; then
            echo "Current default: $DEFAULT_PROFILE"
        else
            echo "No default profile set."
        fi
        echo "Usage: lsync default <profile>"
        return
    fi
    
    local profile_file="$PROFILES_DIR/$profile.conf"
    
    if [[ ! -f "$profile_file" ]]; then
        echo -e "${RED}${CLONE_MARKER}Profile '$profile' does not exist${NC}"
        return 1
    fi
    
    if grep -q "^DEFAULT_PROFILE=" "$CONFIG_FILE" 2>/dev/null; then
        sed -i "s/^DEFAULT_PROFILE=.*/DEFAULT_PROFILE=\"$profile\"/" "$CONFIG_FILE"
    else
        echo "DEFAULT_PROFILE=\"$profile\"" >> "$CONFIG_FILE"
    fi
    
    DEFAULT_PROFILE="$profile"
    echo -e "${GREEN}${CLONE_MARKER}Default profile set to '$profile'${NC}"
}

profile_rename() {
    local old_name="$1"
    local new_name="$2"
    
    if [[ -z "$old_name" || -z "$new_name" ]]; then
        echo "Usage: lsync rename <old-name> <new-name>"
        return 1
    fi
    
    local old_file="$PROFILES_DIR/$old_name.conf"
    local new_file="$PROFILES_DIR/$new_name.conf"
    
    if [[ ! -f "$old_file" ]]; then
        echo -e "${RED}${CLONE_MARKER}Profile '$old_name' does not exist${NC}"
        return 1
    fi
    
    if [[ -f "$new_file" ]]; then
        echo -e "${RED}${CLONE_MARKER}Profile '$new_name' already exists${NC}"
        return 1
    fi
    
    mv "$old_file" "$new_file"
    
    if [[ "$DEFAULT_PROFILE" == "$old_name" ]]; then
        sed -i "s/^DEFAULT_PROFILE=.*/DEFAULT_PROFILE=\"$new_name\"/" "$CONFIG_FILE"
    fi
    
    echo -e "${GREEN}${CLONE_MARKER}Renamed profile '$old_name' to '$new_name'${NC}"
}

load_profile() {
    local profile="$1"
    local profile_file="$PROFILES_DIR/$profile.conf"
    
    if [[ ! -f "$profile_file" ]]; then
        echo -e "${RED}${CLONE_MARKER}Profile '$profile' does not exist${NC}"
        return 1
    fi
    
    source "$profile_file"
    
    return 0
}

get_profile_paths() {
    local profile="$1"
    local profile_file="$PROFILES_DIR/$profile.conf"
    
    if [[ ! -f "$profile_file" ]]; then
        return 1
    fi
    
    grep -oP '^\[\K[^\]]+' "$profile_file" 2>/dev/null
}

profile_sync() {
    local action="$1"
    local profile="$2"
    local path_name="$3"
    
    if [[ -z "$profile" ]]; then
        if [[ -n "$DEFAULT_PROFILE" ]]; then
            profile="$DEFAULT_PROFILE"
        else
            echo -e "${RED}${CLONE_MARKER}No profile specified and no default set${NC}"
            echo "Usage: lsync pull/push <profile> [path-name]"
            echo "Or set a default: lsync default <profile>"
            return 1
        fi
    fi
    
    load_profile "$profile" || return 1
    
    prompt_sudo_if_needed || { echo "Profile sync cancelled."; return 1; }
    
    if [[ -n "$path_name" ]]; then
        local local_val remote_val
        local_val=$(grep -A2 "^\[$path_name\]" "$PROFILES_DIR/$profile.conf" 2>/dev/null | grep "^local=" | cut -d'"' -f2)
        remote_val=$(grep -A2 "^\[$path_name\]" "$PROFILES_DIR/$profile.conf" 2>/dev/null | grep "^remote=" | cut -d'"' -f2)
        
        if [[ -z "$local_val" || -z "$remote_val" ]]; then
            echo -e "${RED}${CLONE_MARKER}Path '$path_name' not found in profile '$profile'${NC}"
            return 1
        fi
        
        local local_dir="$local_val"
        local remote_dir="$remote_val"
        
        if [[ "$QUIET" != "true" ]]; then
            echo -e "${CYAN}${CLONE_MARKER}[$profile/$path_name] Checking...${NC}"
        fi
        
        if [[ "$action" == "push" ]]; then
            if ! get_changes "$local_dir/" "$USER@$HOST:$remote_dir/" "true"; then
                if [[ "$QUIET" != "true" ]]; then
                    echo -e "${GREEN}${CLONE_MARKER}Nothing to commit${NC}"
                fi
                return
            fi
            
            get_changes "$local_dir/" "$USER@$HOST:$remote_dir/" "$QUIET"
            
            if confirm "Push $path_name to server?"; then
                backup_create "$USER@$HOST:$remote_dir/" "$profile_$path_name"
                run_rsync_progress "$local_dir/" "$USER@$HOST:$remote_dir/"
            else
                echo "Push cancelled."
            fi
        else
            if ! get_changes "$USER@$HOST:$remote_dir/" "$local_dir/" "true"; then
                if [[ "$QUIET" != "true" ]]; then
                    echo -e "${GREEN}${CLONE_MARKER}Nothing to commit${NC}"
                fi
                return
            fi
            
            get_changes "$USER@$HOST:$remote_dir/" "$local_dir/" "$QUIET"
            
            if confirm "Pull $path_name from server?"; then
                backup_create "$local_dir/" "$profile_$path_name"
                run_rsync_progress "$USER@$HOST:$remote_dir/" "$local_dir/"
            else
                echo "Pull cancelled."
            fi
        fi
        return
    fi
    
    local paths
    mapfile -t paths < <(get_profile_paths "$profile")
    
    if [[ ${#paths[@]} -eq 0 ]]; then
        echo -e "${YELLOW}${CLONE_MARKER}No paths in profile '$profile'${NC}"
        echo "Add paths with: lsync add $profile <path-name> [folder]"
        return
    fi
    
    for p in "${paths[@]}"; do
        local local_val remote_val
        local_val=$(grep -A2 "^\[$p\]" "$PROFILES_DIR/$profile.conf" 2>/dev/null | grep "^local=" | cut -d'"' -f2)
        remote_val=$(grep -A2 "^\[$p\]" "$PROFILES_DIR/$profile.conf" 2>/dev/null | grep "^remote=" | cut -d'"' -f2)
        
        if [[ -z "$local_val" || -z "$remote_val" ]]; then
            continue
        fi
        
        local local_dir="$local_val"
        local remote_dir="$remote_val"
        
        if [[ "$QUIET" != "true" ]]; then
            echo ""
            echo -e "${CYAN}${CLONE_MARKER}[$profile/$p] Checking...${NC}"
        fi
        
        if [[ "$action" == "push" ]]; then
            if ! get_changes "$local_dir/" "$USER@$HOST:$remote_dir/" "true"; then
                if [[ "$QUIET" != "true" ]]; then
                    echo -e "${GREEN}${CLONE_MARKER}Nothing to commit${NC}"
                fi
                continue
            fi
            
            get_changes "$local_dir/" "$USER@$HOST:$remote_dir/" "$QUIET"
            
            if confirm "Push $p to server?"; then
                backup_create "$USER@$HOST:$remote_dir/" "$profile_$p"
                run_rsync_progress "$local_dir/" "$USER@$HOST:$remote_dir/"
            else
                echo "Push cancelled."
            fi
        else
            if ! get_changes "$USER@$HOST:$remote_dir/" "$local_dir/" "true"; then
                if [[ "$QUIET" != "true" ]]; then
                    echo -e "${GREEN}${CLONE_MARKER}Nothing to commit${NC}"
                fi
                continue
            fi
            
            get_changes "$USER@$HOST:$remote_dir/" "$local_dir/" "$QUIET"
            
            if confirm "Pull $p from server?"; then
                backup_create "$local_dir/" "$profile_$p"
                run_rsync_progress "$USER@$HOST:$remote_dir/" "$local_dir/"
            else
                echo "Pull cancelled."
            fi
        fi
    done
}

run_rsync_progress() {
    local src="$1"
    local dst="$2"
    
    local rsync_path
    rsync_path=$(get_rsync_path)
    local remote_user
    remote_user=$(get_remote_user)
    local dst_with_user
    local filters
    filters=$(build_rsync_filters)
    
    if [[ "$dst" == *@*:* ]]; then
        dst_with_user="${remote_user}@${dst#*@}"
    else
        dst_with_user="$dst"
    fi
    
    local total_files
    total_files=$(rsync -av --stats -n $filters --rsync-path="$rsync_path" -e "ssh -i $SSH_KEY -o StrictHostKeyChecking=no" "$src" "$dst_with_user" 2>&1 | grep -E "Number of regular files transferred" | awk '{print $NF}' | tr -d ',')
    [[ -z "$total_files" ]] && total_files=0
    
    if [[ $total_files -eq 0 ]]; then
        if [[ "$QUIET" != "true" ]]; then
            echo -e "${CYAN}${CLONE_MARKER}Syncing...${NC}"
        fi
        if ! rsync -avz -q $filters --rsync-path="$rsync_path" -e "ssh -i $SSH_KEY -o StrictHostKeyChecking=no" "$src" "$dst_with_user" 2>&1; then
            echo -e "${RED}${CLONE_MARKER}Error: Sync failed${NC}" >&2
            return 1
        fi
        if [[ "$QUIET" != "true" ]]; then
            echo -e "${GREEN}${CLONE_MARKER} Done!${NC}"
        fi
        return
    fi
    
    if [[ "$DEBUG" == true ]]; then
        rsync -avz $filters --rsync-path="$rsync_path" -e "ssh -i $SSH_KEY -o StrictHostKeyChecking=no" "$src" "$dst_with_user"
        return
    fi
    
    local current=0
    local bar_width=30
    local rsync_exit_code=0
    local start_time=$(date +%s)
    local last_time=$start_time
    local bytes_transferred=0
    
    get_progress_color() {
        local pct=$1
        if [[ $pct -lt 33 ]]; then
            echo "${GREEN}"
        elif [[ $pct -lt 66 ]]; then
            echo "${YELLOW}"
        else
            echo "${RED}"
        fi
    }
    
    format_speed() {
        local speed=$1
        if [[ $speed -lt 1024 ]]; then
            echo "${speed}B/s"
        elif [[ $speed -lt 1048576 ]]; then
            echo "$((speed / 1024))KB/s"
        else
            echo "$((speed / 1048576))MB/s"
        fi
    }
    
    format_time() {
        local secs=$1
        local mins=$((secs / 60))
        local rem_secs=$((secs % 60))
        printf "%02d:%02d" $mins $rem_secs
    }
    
    while IFS= read -r line; do
        first_char="${line:0:1}"
        if [[ "$first_char" == ">" || "$first_char" == "<" || "$first_char" == "c" ]]; then
            ((current++))
            local current_time=$(date +%s)
            local elapsed=$((current_time - start_time))
            
            filename=$(echo "$line" | awk '{for(i=5;i<=NF;i++) printf "%s ", $i}' | sed 's/ *$//')
            [[ -z "$filename" ]] && filename="transferring..."
            
            local pct=$((current * 100 / total_files))
            local filled=$((pct * bar_width / 100))
            local empty=$((bar_width - filled))
            
            local speed=0
            if [[ $elapsed -gt 0 ]]; then
                speed=$((current * 50000 / elapsed))
            fi
            
            local eta=0
            if [[ $current -gt 0 && $elapsed -gt 0 ]]; then
                local rate=$((current * 100 / elapsed))
                if [[ $rate -gt 0 ]]; then
                    eta=$(((total_files - current) * elapsed / current))
                fi
            fi
            
            local color
            color=$(get_progress_color $pct)
            
            printf "\r${color}%-${filled}s${RED}%-${empty}s${NC} %3d%% | %s | ETA: %s | %s" \
                "$(printf '#%.0s' $(seq 1 $filled) 2>/dev/null || echo "")" \
                "$(printf '.%.0s' $(seq 1 $empty) 2>/dev/null || echo "")" \
                "$pct" "$(format_speed $speed)" "$(format_time $eta)" "$filename"
        fi
    done < <(rsync -avz $filters --rsync-path="$rsync_path" -e "ssh -i $SSH_KEY -o StrictHostKeyChecking=no" "$src" "$dst_with_user" 2>&1 | grep -v '^sending' | grep -v '^receiving' | grep -v '^total:')
    
    rsync_exit_code=${PIPESTATUS[0]}
    echo ""
    
    if [[ $rsync_exit_code -ne 0 ]]; then
        echo -e "${RED}${CLONE_MARKER}Error: Sync failed with exit code $rsync_exit_code${NC}" >&2
        return 1
    fi
    
    if [[ "$QUIET" != "true" ]]; then
        echo -e "${GREEN}${CLONE_MARKER} Done!${NC}"
    fi
    
    backup_rotate
}

sync_push() {
    local folder="$1"
    local cwd="$(pwd)"
    local local_dir
    local remote_dir
    local backup_name
    
    prompt_sudo_if_needed || { echo "Push cancelled."; return 1; }
    
    if [[ -z "$folder" ]]; then
        local_dir="$cwd"
        remote_dir="$cwd"
        backup_name="$(basename "$cwd")"
    else
        local_dir="$cwd/$folder"
        remote_dir="$cwd/$folder"
        backup_name="$(basename "$folder")"
    fi
    
    if [[ "$QUIET" != "true" ]]; then
        echo -e "${CYAN}${CLONE_MARKER} Checking...${NC}"
    fi
    
    if ! get_changes "$local_dir/" "$USER@$HOST:$remote_dir/" "true"; then
        if [[ "$QUIET" != "true" ]]; then
            echo -e "${GREEN}${CLONE_MARKER} Nothing to commit${NC}"
        fi
        return
    fi
    
    get_changes "$local_dir/" "$USER@$HOST:$remote_dir/" "$QUIET"
    
    if confirm "Are you sure you want to commit those changes?"; then
        if [[ "$QUIET" != "true" ]]; then
            echo "Creating backup of server destination..."
        fi
        backup_create "$USER@$HOST:$remote_dir/" "$backup_name"
        
        run_rsync_progress "$local_dir/" "$USER@$HOST:$remote_dir/"
        if [[ "$QUIET" != "true" ]]; then
            echo -e "${GREEN}${CLONE_MARKER} Done!${NC}"
        fi
    else
        echo "Push cancelled."
    fi
}

main() {
    parse_flags "$@"
    
    if [[ "$COLOR_OUTPUT" == false ]]; then
        RED=''
        GREEN=''
        CYAN=''
        YELLOW=''
        NC=''
    fi
    
    if [[ "$CMD_ACTION" == "init" ]]; then
        if [[ "$CMD_ARG" == "--reset" || -f "$CONFIG_FILE" ]]; then
            if [[ -f "$CONFIG_FILE" && "$CMD_ARG" != "--reset" ]]; then
                echo -e "${YELLOW}${CLONE_MARKER}Config already exists at $CONFIG_FILE${NC}"
                echo -n "Overwrite? [y/N]: "
                read -r confirm
                if [[ "$confirm" != "y" && "$confirm" != "Y" ]]; then
                    echo "Cancelled."
                    exit 0
                fi
            fi
        fi
        init_config
        exit 0
    fi
    
    load_config
    
    if ! validate_config 2>/dev/null; then
        echo "Run 'lsync init' to configure."
        exit 1
    fi
    
    if [[ -z "$CMD_ACTION" ]]; then
        show_help
        exit 0
    fi
    
    case "$CMD_ACTION" in
        push)
            sync_push "${CMD_ARG:-}"
            ;;
        pull)
            sync_pull "${CMD_ARG:-}"
            ;;
        "$PUSH_CMD")
            sync_default "push"
            ;;
        "$PULL_CMD")
            sync_default "pull"
            ;;
        backup|backups)
            backup_list
            ;;
        restore)
            backup_restore "${CMD_ARG:-}"
            ;;
        create)
            profile_create "${CMD_ARG:-}"
            ;;
        add)
            profile_add "${2:-}" "${3:-}" "${4:-}"
            ;;
        remove|rm)
            profile_remove "${2:-}" "${3:-}"
            ;;
        profiles|ls)
            profile_list
            ;;
        default)
            profile_set_default "${2:-}"
            ;;
        rename)
            profile_rename "${2:-}" "${3:-}"
            ;;
        *)
            if [[ -f "$PROFILES_DIR/${CMD_ACTION}.conf" ]]; then
                profile_sync "pull" "$CMD_ACTION" "${CMD_ARG:-}"
            elif [[ "$CMD_ACTION" == "push" && -f "$PROFILES_DIR/${CMD_ARG}.conf" ]]; then
                profile_sync "push" "$CMD_ARG" "${3:-}"
            elif [[ "$CMD_ACTION" == "pull" && -f "$PROFILES_DIR/${CMD_ARG}.conf" ]]; then
                profile_sync "pull" "$CMD_ARG" "${3:-}"
            else
                echo -e "${RED}${CLONE_MARKER}Unknown command: $CMD_ACTION${NC}"
                echo ""
                show_help
                exit 1
            fi
            ;;
    esac
}

main "$@"
